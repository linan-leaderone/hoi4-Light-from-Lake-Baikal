		is_good = no
		selectable_mission = no
		days_mission_timeout = 14

			custom_effect_tooltip = bfi_italy_pick_issue_less_independent_tt

	#2nd mission, all countries need to select a negotiation level. This mission shows up to all nations
	bfi_select_level_mission = {
		allowed = {
			tag = ITA
			tag = USA
			tag = JAP
		}
		available = {
			custom_trigger_tooltip = {
				tooltip = BFI_selected_level_tt

		}
		activation = { always = no } # Will activate by command
		is_good = no
		selectable_mission = no
		days_mission_timeout = 14
		timeout_effect = {
			if = {
				limit = { tag = USA }
				BFI_increase_japan_influence_high = yes
			}
			if = {
				limit = { tag = JAP }
				BFI_increase_usa_influence_high = yes
			}
			if = {
				limit = { tag = ITA }
				BFI_increase_usa_influence_high = yes
				BFI_increase_japan_influence_high = yes

			if = {
				limit = {
					NOT = { ITA = { has_country_flag = bfi_selecting_level } }
					NOT = { USA = { has_country_flag = bfi_selecting_level } }
					NOT = { JAP = { has_country_flag = bfi_selecting_level } }

					USA = { activate_mission = bfi_influence_italy_mission_ITA }
					USA = { activate_mission = bfi_influence_italy_mission_JAP }
					JAP = { activate_mission = bfi_influence_italy_mission_ITA }

				}
			}
		}
		complete_effect = {
			if = {
				limit = {
					NOT = { ITA = { has_country_flag = bfi_selecting_level } }
					NOT = { USA = { has_country_flag = bfi_selecting_level } }
					NOT = { JAP = { has_country_flag = bfi_selecting_level } }
				}
				ITA = { set_country_flag = bfi_influence }

		allowed = { tag = ITA }
		available = { always = no }
		activation = {
			custom_trigger_tooltip = {

				if = {
					limit = { check_variable = { global.bfi_issues_resolved = 5 } }
					clr_country_flag = bfi_active
					clr_country_flag = TNO_BFI_GUI_Italy_Window_visible
					BattleForItaly_checkWinner = yes
					BFI_Italy_joins_winner = yes
				}
				else = {
					USA = { set_variable = { bfi_negotiation_level = -1 } }
					JAP = { set_variable = { bfi_negotiation_level = -1 } }
					ITA = { set_variable = { bfi_negotiation_level = -1 } }
					set_variable = { ITA.bfi_current_issue = 0 }
					activate_mission = bfi_pick_issue_mission

					JAP = { add_to_variable = { bfi_points_available = bfi_point_next_issue } }

					JAP = { set_variable = { bfi_point_next_issue = 2 } }
				}
			}

	#things that get you points
	bfi_gain_points_USA = {
		allowed = { tag = USA }
		visible = {
			ITA = { has_country_flag = bfi_active }
			check_variable = { global.bfi_issues_resolved < 4 }
		}
		available = {
			custom_trigger_tooltip = {
				tooltip = bfi_less_than_4_points_tt
				check_variable = { bfi_point_next_issue < 4 }
			}
		}
		ai_will_do = {
			base = 0
			modifier = {
				add = 1
				check_variable = { bfi_point_next_issue = 2 }
			}
		}
		cost = 60
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision bfi_gain_points_USA points at [?bfi_point_next_issue]"
			custom_effect_tooltip = bfi_gain_points_tt
			if  = {
				limit = {
					check_variable = { bfi_point_next_issue = 3 }
				}
				if = {
					limit = { USA_RD_ruling_party = yes }
					hidden_effect = {
						every_owned_state = {
							limit = { can_have_elections_in_state = yes }
							add_to_variable = { NPP_popularity = 0.025 }
							subtract_from_variable = { RD_popularity = 0.025 }
						}
					}
					custom_effect_tooltip = USA_RD_popularity_decrease_low
				}
				else = {
					custom_effect_tooltip = USA_NPP_popularity_decrease_low
					hidden_effect = {
						every_owned_state = {
							limit = { can_have_elections_in_state = yes }
							add_to_variable = { RD_popularity = 0.025 }
							subtract_from_variable = { NPP_popularity = 0.025 }
						}
					}
				}
			}
			add_to_variable = { bfi_point_next_issue = 1 }
		}
	}
