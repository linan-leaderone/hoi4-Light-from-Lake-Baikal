	MEX_kabuki_crisis_continues = {
		icon = GFX_decision_generic_clock
		cancel_trigger = { hidden_trigger = { has_country_flag = mateos_final_tree_flag } }
		activation = {
			OR = {
				has_completed_focus = MEX_rallying_the_party
				has_completed_focus = MEX_the_technocratic_counter
			}
		}
		available = {
			custom_trigger_tooltip = {
				tooltip = MEX_blank_tooltip
				NOT = { has_country_flag = MEX_active_crisis }
			}
		}
		fire_only_once = no
		selectable_mission = no
		days_mission_timeout = 10
		is_good = no
		priority = 100000

		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: decision timeout MEX_kabuki_crisis_continues"
			add_stability = -0.005
			if = {
				limit = { NOT = { has_country_flag = MEX_kabuki_pressured_the_governors } }
				MEX_PRI_decrease_ordaz_pop_officials_very_low = yes
				MEX_PRI_decrease_madrazo_pop_officials_very_low = yes
			}
			if = {
				limit = { NOT = { has_country_flag = MEX_kabuki_tempered_investors_fears } }
				MEX_PRI_decrease_lozano_pop_businessmen_very_low = yes
			}
			MEX_PRI_decrease_madrazo_pop_workers_very_low = yes
			MEX_PRI_decrease_ordaz_pop_workers_very_low = yes
			MEX_PRI_decrease_loyalty_all_factions_very_low = yes
			if = {
				limit = { NOT = { has_country_flag = MEX_kabuki_pressure_groups_formed } }
				MEX_SB_low_decrease_unity = yes
			}
			hidden_effect = {
				if = {
					limit = { check_variable = { MEX_PRI_kabuki_effect_tension < 21 } }
					MEX_PRI_decrease_lozano_pop_all_factions_low = yes
				}
				else_if = {
					limit = { check_variable = { MEX_PRI_kabuki_effect_tension < 36 } }
					MEX_PRI_decrease_lozano_pop_all_factions_very_low = yes
				}
				else_if = {
					limit = { check_variable = { MEX_PRI_kabuki_effect_tension < 65 } }
				}
				else_if = {
					limit = { check_variable = { MEX_PRI_kabuki_effect_tension < 80 } }
					MEX_PRI_increase_lozano_pop_all_factions_very_low = yes
				}
				else = {
					MEX_PRI_increase_lozano_pop_all_factions_low = yes
				}
				if = {
					limit = { check_variable = { MEX_PRI_kabuki_effect_pact_strength < 21 } }
					MEX_PRI_decrease_madrazo_pop_all_factions_low = yes
					MEX_PRI_increase_ordaz_pop_all_factions_low = yes
				}
				else_if = {
					limit = { check_variable = { MEX_PRI_kabuki_effect_pact_strength < 36 } }
					MEX_PRI_decrease_madrazo_pop_all_factions_very_low = yes
					MEX_PRI_increase_ordaz_pop_all_factions_very_low = yes
				}
				else_if = {
					limit = { check_variable = { MEX_PRI_kabuki_effect_pact_strength < 65 } }
				}
				else_if = {
					limit = { check_variable = { MEX_PRI_kabuki_effect_pact_strength < 80 } }
					MEX_PRI_decrease_ordaz_pop_all_factions_very_low = yes
					MEX_PRI_increase_madrazo_pop_all_factions_very_low = yes
				}
				else = {
					MEX_PRI_decrease_ordaz_pop_all_factions_low = yes
					MEX_PRI_increase_madrazo_pop_all_factions_low = yes
				}
			}
			if = {
				limit = { check_variable = { MEX_PRI_crisis_number = 1 } }
				activate_mission = MEX_kabuki_crisis_continues
			}
		}
	}
