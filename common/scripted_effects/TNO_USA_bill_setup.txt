USA_bill_setup = { #some helpful functions. this one applys the discounts from having a unified party
	US_elections_calculate_country_vars = yes
	USA_GUI_total_up_senators = yes
	if = {
		limit = {
			OR = {
				has_government = liberalism
				has_government = conservatism
			}
		}
		set_temp_variable = { temp_D_unity_bonus = USA_LD_Coalition_Sup }
		multiply_temp_variable = { temp_D_unity_bonus = 3 }
		add_to_temp_variable = { temp_D_unity_bonus = USA_DC_Coalition_Sup }
		multiply_temp_variable = { temp_D_unity_bonus = 45 }
		divide_temp_variable = { temp_unity_bonus = 4 }
		subtract_from_variable = { RD_D_flip_cost = temp_D_unity_bonus } #you get a pp discount based on unity
		
		set_temp_variable = { temp_R_unity_bonus = USA_RR_Coalition_Sup }
		add_to_temp_variable = { temp_R_unity_bonus = USA_HR_Coalition_Sup }
		multiply_temp_variable = { temp_R_unity_bonus = 45 }
		divide_temp_variable = { temp_R_unity_bonus = 2 }
		subtract_from_variable = { RD_R_flip_cost = temp_R_unity_bonus } #you get a pp discount based on unity
	}
	else_if = {
		limit = {
			OR = {
				has_government = paternalism
				has_government = progressivism
				has_government = national_socialism
				has_government = communist
			}
		}
		set_temp_variable = { temp_unity_bonus = USA_NPP_Unity }
		multiply_temp_variable = { temp_unity_bonus = 5 }
		if = {
			limit = { has_variable = C_NPP_flip_cost }
			subtract_from_variable = { C_NPP_flip_cost = temp_unity_bonus } #you get a pp discount based on unity
		}
		if = {
			limit = { has_variable = L_NPP_flip_cost }
			subtract_from_variable = { L_NPP_flip_cost = temp_unity_bonus }
		}
		if = {
			limit = { has_variable = FR_NPP_flip_cost }
			subtract_from_variable = { FR_NPP_flip_cost = temp_unity_bonus }
		}
		if = {
			limit = { has_variable = Y_NPP_flip_cost }
			subtract_from_variable = { Y_NPP_flip_cost = temp_unity_bonus }
		}
	}
	USA_initialize_bill_support = yes
	USA_Dynamic_Bill_Support = yes
}

check_if_bill_passed = {
	set_variable = { percent_support = supporting_senators }
	divide_variable = { percent_support = total_senators }
	set_variable = { votes_needed = 0.5 }
	if = {
		limit = { has_country_flag = fillibuster_active }
		set_variable = { votes_needed = 0.6 }
	}

	if = {
		limit = {
			OR = {
				check_variable = {
					var = percent_support
					value = votes_needed
					compare = greater_than_or_equals
				}
				AND = {
					is_ai = yes
					has_game_rule = {
						rule = USA_BILL_PASS_CHEAT_1964
						option = USA_BILL_PASS_CHEAT_1964_YES
					} 
					date < 1969.1.1
				}
				AND = {
					is_ai = yes
					has_game_rule = {
						rule = USA_BILL_PASS_CHEAT_1968
						option = USA_BILL_PASS_CHEAT_1968_YES
					} 
					date > 1968.10.3
					date < 1973.11.1
				}
				AND = {
					is_ai = yes
					has_game_rule = {
						rule = USA_BILL_PASS_CHEAT_1972
						option = USA_BILL_PASS_CHEAT_1972_YES
					} 
					date > 1972.10.3
					date < 1977.1.1
				}
				AND = {
					OR = {
						has_game_rule = {
							rule = US_JAPAN_DETENTE_PHASE_ONE
							option = US_JAPAN_DETENTE_PHASE_ONE_SUCCESS
						}
						has_game_rule = {
							rule = US_JAPAN_DETENTE_PHASE_ONE
							option = US_JAPAN_DETENTE_PHASE_ONE_FULL_SUCCESS
						}
					}
					OR = {
						has_country_flag = USA_kissinger_popular_treaty_decision_lol
						has_country_flag = USA_kissinger_unpopular_treaty_decision_lol
					}
					NOT = {
						has_country_flag = USA_Hart_kissinger_treaty_1_decision
					}
				}
				AND = {
					OR = {
						has_game_rule = {
							rule = US_JAPAN_DETENTE_PHASE_TWO
							option = US_JAPAN_DETENTE_PHASE_TWO_SUCCESS
						}
						has_game_rule = {
							rule = US_JAPAN_DETENTE_PHASE_TWO
							option = US_JAPAN_DETENTE_PHASE_TWO_FULL_SUCCESS
						}
					}
					has_country_flag = USA_kissinger_PAST
				}
			}
		}
		set_country_flag = bill_passed
	}
}
