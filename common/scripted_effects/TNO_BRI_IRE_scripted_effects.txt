IRE_mog_popularity_measure_positive = {
	if = {
		limit = { check_variable = { mog_temp > 5 } }
		custom_effect_tooltip = IRE_gain_app_big.tt
	}
	else_if = {
		limit = { AND = {
			check_variable = { mog_temp > 0 }
			check_variable = { mog_temp < 5 }
		} }
		custom_effect_tooltip = IRE_gain_app_small.tt
	}
	add_to_variable = { mog_app = mog_temp }
	IRE_calc_mog_app_pp = yes
}

IRE_mog_popularity_measure_negative = {
	if = {
		limit = { check_variable = { mog_temp > 10 } }
		custom_effect_tooltip = IRE_lose_app_devastating.tt
	}
	else_if = {
		limit = { AND = {
			check_variable = { mog_temp > 5 }
			check_variable = { mog_temp < 10 }
		} }
		custom_effect_tooltip = IRE_lose_app_big.tt
	}
	else_if = {
		limit = { AND = {
			check_variable = { mog_temp > 0 }
			check_variable = { mog_temp < 5 }
		} }
		custom_effect_tooltip = IRE_lose_app_med.tt
	}
	else_if = {
		limit = { check_variable = { mog_temp > 0 } }
		custom_effect_tooltip = IRE_lose_app_small.tt
	}
	subtract_from_variable = { mog_app = mog_temp }
	IRE_calc_mog_app_pp = yes
}

IRE_calc_mog_app_pp = {
	set_variable = { mog_app_pp = mog_app }
	subtract_from_variable = { mog_app_pp = 50 }
	divide_variable = { mog_app_pp = 100 }
	force_update_dynamic_modifier = yes
}
