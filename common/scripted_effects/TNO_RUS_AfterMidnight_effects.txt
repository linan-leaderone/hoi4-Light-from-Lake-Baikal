TNO_AM_country_setup = {
	load_focus_tree = ZZZ_blank_focus
	every_owned_state = {
		add_core_of = PREV
		set_demilitarized_zone = no
	}
	set_country_flag = RUS_cancel_all_ideas
	remove_all_ministers = yes
	if = {
		limit = { NOT = { tag = KOM } }
		inherit_technology = KOM
	}
	add_ideas = {
		AM_salted_earth
		AM_Empty_hog
		AM_Empty_for
		AM_Empty_eco
		AM_Empty_sec
	}
	meta_effect = {
		text = {
			load_oob = "[AMTag]_AM"
		}
		AMTag = "[THIS.GetTag]"
	}

	set_temp_variable = { temp_credit_rating = 3 }
	set_temp_variable = { temp_credit_rating_min = 1 }
	set_temp_variable = { temp_credit_rating_max = 4 }
	econ_initialize_credit_rating_system = yes
	set_variable = { poverty_rate = 85 }

	set_variable = { loot_amount = 1 }
	set_country_flag = is_AM_warlord
	set_country_flag = is_warlord
	clr_country_flag = RUS_regional_stage
	clr_country_flag = RUS_superregional_stage
	clr_country_flag = RUS_cannot_raid
	clr_country_flag = russia_preparing_raid
	clr_country_flag = russia_is_raid_target
}
