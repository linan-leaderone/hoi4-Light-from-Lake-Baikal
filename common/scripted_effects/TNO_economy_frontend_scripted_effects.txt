# Should be used VERY rarely and only for significant events
TNO_change_poverty_rate_direct = {
	add_to_variable = { poverty_rate = temp_poverty_change }
	clamp_variable = { var = poverty_rate min = 0 max = 100 }
	if = {
		limit = {
			check_variable = { temp_poverty_change < 0 }
		}
		multiply_temp_variable = {
			var = temp_poverty_change
			value = -1
		}
		custom_effect_tooltip = improve_poverty_rate_direct_tt
	}
	else = {
		custom_effect_tooltip = worsen_poverty_rate_direct_tt
	}
}

econ_spend_money_once_effect_raw_money = { #Spend temp_econ_spending_amount billion dollars with tooltip (0,3 = 300 million $)
	set_temp_variable = { temp_econ_spending_amount_unmodified = temp_econ_spending_amount }
	if = {
		limit = {
			check_variable = {
				var = temp_econ_spending_amount
				value = money_reserves
				compare = less_than_or_equals
			}
		}
		subtract_from_variable = { money_reserves = temp_econ_spending_amount }
		custom_effect_tooltip = TNO_spend_money_tt
	}
	else_if = {
		limit = {
			check_variable = { temp_econ_spending_amount_unmodified < 0 }
		}
		set_temp_variable = { money_reserves_temp = temp_econ_spending_amount_unmodified }
		multiply_temp_variable = { money_reserves_temp = -1 }
		econ_money_reserves_change_raw_money = yes
	}
	else = {
		subtract_from_temp_variable = { temp_econ_spending_amount = money_reserves }
		set_variable = { money_reserves = 0 }
		add_to_variable = { national_debt = temp_econ_spending_amount }
		custom_effect_tooltip = TNO_spend_money_tt
	}
	update_economy_tab = yes
	clear_variable = temp_econ_spending_amount
}

econ_money_reserves_change_raw_money = { #Uses Temp Vars to make it a lot simpler + no mass tooltip spam, value of parameter in billions
	add_to_variable = { money_reserves = money_reserves_temp }
	if = {
		limit = { is_ai = yes }
		if = {
			limit = {
				check_variable = {
					var = money_reserves 
					value = national_debt 
					compare = less_than_or_equals
				}
			}
			subtract_from_variable = { national_debt = money_reserves }
			set_variable = { money_reserves = 0 }
		}
		else = {
			subtract_from_variable = { money_reserves = national_debt }
			set_variable = { national_debt = 0 }
		}
		update_economy_tab = yes
	}

	if = {
		limit = { check_variable = { money_reserves_temp < 0 } }
		multiply_temp_variable = { money_reserves_temp = -1 }
		custom_effect_tooltip = TNO_econ_money_reserves_decrease_tt
	} else = {
		custom_effect_tooltip = TNO_econ_money_reserves_increase_tt
	}
	calculate_display_variables = yes
}

stockpile_chromium_change = {
	add_to_variable = { TNO_stockpile^6 = stockpile_temp }
	if = {
		limit = {
			check_variable = { stockpile_temp < 0 }
		}
		multiply_temp_variable = {
			var = stockpile_temp
			value = -1
		}
		custom_effect_tooltip = TNO_stockpile_chromium_decrease_tt
	}
	else = {
		custom_effect_tooltip = TNO_stockpile_chromium_increase_tt
	}
	clamp_variable = { var = TNO_stockpile^6 min = 0 max = 9999 }
}
