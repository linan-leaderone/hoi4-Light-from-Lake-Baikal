SCO_English_Minority_INIT = {

	set_variable = { SCO_English_Minority_pp = -0.46	}
	set_variable = { SCO_English_Minority_stab = -0.24	}
	set_variable = { SCO_English_Minority_ws = -0.19	}
	set_variable = { SCO_English_Minority_icf = -0.14	}
	set_variable = { SCO_English_Minority_cf = -0.26	}
	add_dynamic_modifier = { modifier = SCO_English_Minority }
}

SCO_English_Minority_CALC = {
	clamp_variable = {
		var = SCO_English_Approval
		min = 0
		max = 100
	}
	clamp_variable = {
		var = SCO_English_Population_Percent
		min = 0
		max = 100
	}
	clamp_variable = {
		var = SCO_English_Poverty_Percent_c
		min = 0
		max = 100
	}
	#math for recruitable population change 
	set_temp_variable = {
		SCO_approval_temp = SCO_English_Approval
	}
	set_temp_variable = {
		SCO_pop_temp = SCO_English_Population_Percent
	}
	set_temp_variable = {
		SCO_test_number = 100
	}
	subtract_from_temp_variable = {
		SCO_test_number = SCO_approval_temp #100-24 = 76
	}
	divide_temp_variable = {
		SCO_test_number = 100 # 0.76
	}
	multiply_temp_variable = {
		SCO_pop_temp = SCO_test_number
	}
	set_temp_variable = {
		SCO_pp_temp = SCO_test_number #0.76
	}
	multiply_temp_variable = {
		SCO_pop_temp = -1
	}
	divide_temp_variable = {
		SCO_pop_temp = 100
	}
	set_variable = {
		SCO_English_Minority_cf = SCO_pop_temp
	}

	#set pp loss/gain (approval)
	set_temp_variable = {
		SCO_test_number = 100
	}
	subtract_from_temp_variable = {
		SCO_test_number = SCO_English_Approval #100-24 = 76
	}

	set_temp_variable = {
		SCO_pp_max = 30
	}
	subtract_from_temp_variable = {
		SCO_pp_max = SCO_test_number #60-76 = -16
	}
	divide_temp_variable = {
		SCO_pp_max = 100 #-0.16
	}

	set_variable = {
		SCO_English_Minority_pp = SCO_pp_max
	}

	#set production loss (poverty and pop percent)
	set_temp_variable = {
		SCO_poverty_temp = SCO_English_Poverty_Percent #28
	}
	set_temp_variable = {
		SCO_pop_temp = SCO_English_Population_Percent #35
	}
	divide_temp_variable = {
		SCO_poverty_temp = 100 #0.28
	}
	multiply_temp_variable = {
		SCO_pop_temp = SCO_poverty_temp #35 x 0.28  = 9.8%
	}
	divide_temp_variable = {
		SCO_pop_temp = 100
	}
	multiply_temp_variable = {
		SCO_pop_temp = -1.5
	}
	set_variable = {
		SCO_English_Minority_icf = SCO_pop_temp
	}

	###########################set war sapport loss (simple approval)
	set_temp_variable = {
		SCO_approval_temp = SCO_English_Approval  #24
	}
	divide_temp_variable = {
		SCO_approval_temp = 2 #12
	}
	set_temp_variable = {
		SCO_negative_test = SCO_English_Population_Percent #35
	}
	multiply_temp_variable = {
		SCO_negative_test = -0.9 #-35
	}
	add_to_temp_variable = {
		SCO_negative_test = SCO_approval_temp #-23 ws
	}
	divide_temp_variable = {
		SCO_negative_test = 100 #-0.23
	}


	set_variable = { SCO_English_Minority_ws = SCO_negative_test }

	#calculate stability######################################################################################
	set_temp_variable = {
		SCO_approval_temp = SCO_English_Approval  #24
	}
	divide_temp_variable = {
		SCO_approval_temp = 2 #12
	}
	set_temp_variable = {
		SCO_negative_test = SCO_English_Population_Percent #35
	}
	multiply_temp_variable = {
		SCO_negative_test = -0.9 #-35
	}
	set_temp_variable = {
		SCO_poverty_temp = SCO_English_Poverty_Percent #poverty #28
	}
	divide_temp_variable = {
		SCO_poverty_temp = 100 #0.28
	}
	add_to_temp_variable = {
		SCO_poverty_temp = 1 #1.28
	}
	add_to_temp_variable = {
		SCO_negative_test = SCO_approval_temp #-23 ws
	}
	multiply_temp_variable = {
		SCO_negative_test = SCO_poverty_temp
	}
	divide_temp_variable = {
		SCO_negative_test = 100 #-0.23
	}


	set_variable = { SCO_English_Minority_stab = SCO_negative_test }
}
