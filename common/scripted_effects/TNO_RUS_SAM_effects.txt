	clamp_variable = { var = SAM_democrat_faction_strength min = 2.5 max = 95 }
	clamp_variable = { var = SAM_business_faction_strength min = 2.5 max = 95 }
	clamp_variable = { var = SAM_ROA_faction_strength min = 2.5 max = 95 }


	## Temp Variable Definition
	set_temp_variable = { A = SAM_democrat_faction_strength }
	set_temp_variable = { B = SAM_business_faction_strength }
	set_temp_variable = { C = SAM_ROA_faction_strength }
	# ABC = A+B+C
	set_temp_variable = { ABC = A }
	add_to_temp_variable = { ABC = B }
	add_to_temp_variable = { ABC = C }

	## Calc X
	# 1/2 * ((2B+C)/(A+B+C))
	set_variable = { Zykovtangle_calc_X = B }
	multiply_variable = { Zykovtangle_calc_X = 2 }
	add_to_variable = { Zykovtangle_calc_X = C }
	divide_variable = { Zykovtangle_calc_X = ABC }	#A+B+C
	multiply_variable = { Zykovtangle_calc_X = 0.5 }	#1/2

	## Calc Y
	# sqrt(3)/2 * ((C)/(A+B+C))
	set_variable = { Zykovtangle_calc_Y = C }
	divide_variable = { Zykovtangle_calc_Y = ABC }	#A+B+C
	multiply_variable = { Zykovtangle_calc_Y = 0.866 }	#sqrt(3)/2

	## Scaling
	multiply_variable = { Zykovtangle_calc_X = 496 }
	multiply_variable = { Zykovtangle_calc_Y = 392 }
	multiply_variable = { Zykovtangle_calc_Y = -1 }
	add_to_variable = { Zykovtangle_calc_Y = 392 }

	log = "[GetDateText]: [Root.GetName]: Zykovtangle_calc_XY; Completed execution."
}

SAM_change_zykov_dominant_faction_spirit = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { SAM_ROA_faction_strength > SAM_business_faction_strength }
				check_variable = { SAM_ROA_faction_strength > SAM_democrat_faction_strength }
			}
			remove_ideas = {
				SAM_zykov_business_dominant
				SAM_zykov_democrats_dominant
			}
			add_ideas = SAM_zykov_ROA_dominant
		}
		else_if = {
			limit = {
				check_variable = { SAM_business_faction_strength > SAM_ROA_faction_strength }
				check_variable = { SAM_business_faction_strength > SAM_democrat_faction_strength }
			}
			remove_ideas = {
				SAM_zykov_democrats_dominant
				SAM_zykov_ROA_dominant
			}
			add_ideas = SAM_zykov_business_dominant
		}
		else_if = {
			limit = {
				check_variable = { SAM_democrat_faction_strength > SAM_ROA_faction_strength }
				check_variable = { SAM_democrat_faction_strength > SAM_business_faction_strength }
			}
			remove_ideas = {
				SAM_zykov_business_dominant
				SAM_zykov_ROA_dominant
			}
			add_ideas = SAM_zykov_democrats_dominant
		}
		else_if = {
			limit = {
				OR = {
					check_variable = { SAM_democrat_faction_strength = SAM_ROA_faction_strength }
					check_variable = { SAM_democrat_faction_strength = SAM_business_faction_strength }
					check_variable = { SAM_ROA_faction_strength = SAM_business_faction_strength }
				}
			}
			remove_ideas = {
				SAM_zykov_business_dominant
				SAM_zykov_ROA_dominant
				SAM_zykov_democrats_dominant
			}
		}
	}
}
