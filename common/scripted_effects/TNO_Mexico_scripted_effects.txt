#	1.4.0 Interconnecting Effects

MEX_pressure_coercion_connecting = {
	add_to_variable = { MEX_madrazo_mech_dirty = 0.01 }
	if = { limit = { check_variable = { MEX_forge_USA_tensions > 79 } }
		set_variable = { MEX_forge_USA_pressure = 5 }
	}
	else_if = { limit = { check_variable = { MEX_forge_USA_tensions > 59 } }
		set_variable = { MEX_forge_USA_pressure = 4 }
	}
	else_if = { limit = { check_variable = { MEX_forge_USA_tensions > 39 } }
		set_variable = { MEX_forge_USA_pressure = 3 }
	}
	else_if = { limit = { check_variable = { MEX_forge_USA_tensions > 19 } }
		set_variable = { MEX_forge_USA_pressure = 2 }
	}
	else_if = { set_variable = { MEX_forge_USA_pressure = 1 } }
	if = { limit = { check_variable = { MEX_forge_JAP_tensions > 79 } }
		set_variable = { MEX_forge_JAP_pressure = 5 }
	}
	else_if = { limit = { check_variable = { MEX_forge_JAP_tensions > 59 } }
		set_variable = { MEX_forge_JAP_pressure = 4 }
	}
	else_if = { limit = { check_variable = { MEX_forge_JAP_tensions > 39 } }
		set_variable = { MEX_forge_JAP_pressure = 3 }
	}
	else_if = { limit = { check_variable = { MEX_forge_JAP_tensions > 19 } }
		set_variable = { MEX_forge_JAP_pressure = 2 }
	}
	else_if = { set_variable = { MEX_forge_JAP_pressure = 1 } }
	MEX_forge_of_the_revolution_calculate_gui_effects = yes
}

#	1.5.0 Calculating Gui Effects

MEX_forge_of_the_revolution_calculate_gui_effects = {
	add_to_variable = { MEX_madrazo_mech_dirty = 0.01 }

	clear_array = MEX_forge_of_the_revolution_effects_array
	add_to_array = { MEX_forge_of_the_revolution_effects_array = 0 }
	resize_array = { MEX_forge_of_the_revolution_effects_array = 14 }

	# Political Power
	set_variable = { MEX_forge_of_the_revolution_effects_array^0 = 0.1 }
	multiply_variable = { MEX_forge_of_the_revolution_effects_array^0 = MEX_revolutionary_nationalism }

	# Stability
	set_variable = { MEX_forge_of_the_revolution_effects_array^1 = 0.02 }
	multiply_variable = { MEX_forge_of_the_revolution_effects_array^1 = MEX_revolutionary_nationalism }
	set_variable = { MEX_forge_of_the_revolution_effects_array^11 = -0.04 }
	multiply_variable = { MEX_forge_of_the_revolution_effects_array^11 = MEX_forge_USA_pressure }
	set_variable = { MEX_forge_of_the_revolution_effects_array^12 = MEX_forge_of_the_revolution_effects_array^1 }
	add_to_variable = { MEX_forge_of_the_revolution_effects_array^12 = MEX_forge_of_the_revolution_effects_array^11 }

	# War Support
	set_variable = { MEX_forge_of_the_revolution_effects_array^2 = 0.03 }
	multiply_variable = { MEX_forge_of_the_revolution_effects_array^2 = MEX_revolutionary_nationalism }

	# Peasant/Worker Loyalty/Power
	set_variable = { MEX_forge_of_the_revolution_effects_array^3 = 2 }
	multiply_variable = { MEX_forge_of_the_revolution_effects_array^3 = MEX_revolutionary_nationalism }

	# Trade Opinion Factor
	set_variable = { MEX_forge_of_the_revolution_effects_array^4 = -0.03 }
	multiply_variable = { MEX_forge_of_the_revolution_effects_array^4 = MEX_forge_USA_pressure }
	set_variable = { MEX_forge_of_the_revolution_effects_array^5 = -0.02 }
	multiply_variable = { MEX_forge_of_the_revolution_effects_array^5 = MEX_forge_JAP_pressure }
	set_variable = { MEX_forge_of_the_revolution_effects_array^6 = MEX_forge_of_the_revolution_effects_array^4 }
	add_to_variable = { MEX_forge_of_the_revolution_effects_array^6 = MEX_forge_of_the_revolution_effects_array^5 }

	# Inflation Rate
	set_variable = { MEX_forge_of_the_revolution_effects_array^7 = 0.2 }
	multiply_variable = { MEX_forge_of_the_revolution_effects_array^7 = MEX_forge_JAP_pressure }

	# Interest Rate
	set_variable = { MEX_forge_of_the_revolution_effects_array^8 = 0.2 }

	set_variable = { MEX_forge_of_the_revolution_effects_array^9 = 0.2 }
	multiply_variable = { MEX_forge_of_the_revolution_effects_array^9 = MEX_forge_JAP_pressure }

# 2.0.0 Kabuki Effect - Faction's Loyalty
MEX_kabuki_effect_businessmen_support = {
	if = {
		limit = {
			has_country_flag = MEX_kabuki_industralists_pick_ordaz
		}
		set_variable = { crisis_support_icon_businessmen_var = 5 }
	}
	else_if = {
		limit = {
			has_country_flag = MEX_kabuki_industralists_pick_lozano
		}
		set_variable = { crisis_support_icon_businessmen_var = 6 }
	}
	else_if = {
		limit = {
			has_country_flag = MEX_kabuki_industralists_pick_madrazo
		}
		set_variable = { crisis_support_icon_businessmen_var = 7 }
	}
	else_if = {
		limit = {
			check_variable = { MEX_PRI_group_ordaz_pop^4 > MEX_PRI_group_lozano_pop^4 }
			check_variable = { MEX_PRI_group_ordaz_pop^4 > MEX_PRI_group_madrazo_pop^4 }
		}
		set_variable = { crisis_support_icon_businessmen_var = 2 }
	}
	else_if = {
		limit = {
			check_variable = { MEX_PRI_group_lozano_pop^4 > MEX_PRI_group_ordaz_pop^4 }
			check_variable = { MEX_PRI_group_lozano_pop^4 > MEX_PRI_group_madrazo_pop^4 }
		}
		set_variable = { crisis_support_icon_businessmen_var = 3 }
	}
	else_if = {
		limit = {
			check_variable = { MEX_PRI_group_madrazo_pop^4 > MEX_PRI_group_ordaz_pop^4 }
			check_variable = { MEX_PRI_group_madrazo_pop^4 > MEX_PRI_group_lozano_pop^4 }

MEX_PRI_process_update = {
	clamp_variable = { var = MEX_PRI_group_loyalty^0 max = 100 min = 0}

	for_loop_effect = {
		start = 0
		end = 6
		value = i

		clamp_variable = { var = MEX_PRI_group_loyalty^i max = 100 min = 0}
		clamp_variable = { var = MEX_PRI_group_power^i max = 100 min = 0}
		clamp_variable = { var = MEX_PRI_group_support^i max = 100 min = 0}
		clamp_variable = { var = MEX_PRI_group_corruption^i max = 100 min = 0 }

		clamp_variable = { var = MEX_PRI_group_lozano_pop^i max = 100 min = 0}
		clamp_variable = { var = MEX_PRI_group_madrazo_pop^i max = 100 min = 0}
		clamp_variable = { var = MEX_PRI_group_ordaz_pop^i max = 100 min = 0}

		set_temp_variable = { total = MEX_PRI_group_lozano_pop^i }
		add_to_temp_variable = { total = MEX_PRI_group_madrazo_pop^i }
		add_to_temp_variable = { total = MEX_PRI_group_ordaz_pop^i }
		divide_temp_variable = { total = 100 }

		divide_variable = { MEX_PRI_group_lozano_pop^i = total }
		divide_variable = { MEX_PRI_group_madrazo_pop^i = total }
		divide_variable = { MEX_PRI_group_ordaz_pop^i = total }
	}
}
