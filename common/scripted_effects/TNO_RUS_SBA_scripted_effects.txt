	}

}
SBA_collapse_update = {
	custom_effect_tooltip = SBA_collapse_counter_change_tt
	add_to_variable = { SBA_collapse_counter = SBA_collapse_counter_change }
	clamp_variable = {
		var = SBA_collapse_counter
		min = 0
		max = 100
	}
	if = { 
		limit = {
			has_game_rule = {
				rule = SBA_restoring_order_path
				option = SBA_RESTORING_ORDER_PATH_COLLAPSE
			}
		}
		set_variable = { SBA_collapse_counter = 100 }
	}
	else_if = { 
		limit = {
			has_game_rule = {
				rule = SBA_restoring_order_path
				option = SBA_RESTORING_ORDER_PATH_STABLE
			}
		}
		set_variable = { SBA_collapse_counter = 0 }
	}
	if = {
		limit = {
			check_variable = { SBA_collapse_counter > 79 }
		}
		set_variable = { SBA_collapse_stage = 5 }
	}
	else_if = {
		limit = {
			check_variable = { SBA_collapse_counter > 59 }
		}
		set_variable = { SBA_collapse_stage = 4 }
	}
	else_if = {
		limit = {
			check_variable = { SBA_collapse_counter > 39 }
		}
		set_variable = { SBA_collapse_stage = 3 }
	}
	else_if = {
		limit = {
			check_variable = { SBA_collapse_counter > 19 }
		}
		set_variable = { SBA_collapse_stage = 2 }
	}
	else = {
		set_variable = { SBA_collapse_stage = 1 }
	}
	hidden_effect = {
		if = {
			limit = {
				check_variable = { SBA_collapse_counter = 100 }
				NOT = { has_country_flag = SBA_collapse_flag }
			}
			set_country_flag = SBA_collapse_flag
			if = {
				limit = {
					has_country_flag = SBA_kostinsty_rejected
				}
				country_event = {
					id = SBA.94
					days = 3
				}
			}
			else = {
				country_event = {
					id = SBA.96
					days = 3
				}
			}
		}
	}
