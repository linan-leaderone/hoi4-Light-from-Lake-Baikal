		array = tno_policy_effectiveness_max_array
		value = max_value #while i'm here, might as well save some time
		index = i

		set_temp_variable = { effect = max_value }
		subtract_from_temp_variable = { effect = tno_policy_effectiveness_min_array^i }
		multiply_temp_variable = { effect = tno_policy_effectiveness_temp_math }		#defined in prev scope
		add_to_temp_variable = { effect = tno_policy_effectiveness_min_array^i }

		add_to_array = { tno_policy_effectiveness_effect_array = effect }
	}

	clear_array = tno_policy_effectiveness_max_array
	clear_array = tno_policy_effectiveness_min_array
