defined_text = {
	name = MEX_DW_get_units_recurse
	text = {
		trigger = {
			AND = {
				check_variable = { holder < MEX_DW_node_units^num }
				var:MEX_DW_node_units^holder = {
					AND = {
						check_variable = { MEX_DW_alliegance = token:MEX_Faction_GOVERNMENT }
						check_variable = { MEX_DW_UNIT_TYPE = TYPE }
					}
				}
			}
			var:MEX_DW_node_units^holder = {
				add_to_temp_variable = { recurse = MEX_DW_node_num_units }
			}
			add_to_temp_variable = { holder = 1 }
		}
		localization_key = "[MEX_DW_get_units_recurse]"
	}
	text = {
		trigger = {
			check_variable = { holder < MEX_DW_node_units^num }
			add_to_temp_variable = { holder = 1 }
		}
		localization_key = "[MEX_DW_get_units_recurse]"
	}
	text = {
		localization_key = "[?recurse]"
	}
