			set_temp_variable = { min_num_senators = current_support_pct }

			multiply_temp_variable = { max_num_senators = multiplier }
			multiply_temp_variable = { min_num_senators = num_senators }
			round_temp_variable = min_num_senators
			add_to_temp_variable = { min_num_senators = modifier }
			clamp_temp_variable = { var = min_num_senators min = 0 max = num_senators }
			subtract_from_temp_variable = { min_num_senators = current_support_raw }

			set_temp_variable = { max_num_senators = current_support_pct }
			set_temp_variable = { multiplier = 1 }
			add_to_temp_variable = { multiplier = max_amnt }
			multiply_temp_variable = { max_num_senators = multiplier }

			add_to_temp_variable = { max_num_senators = modifier }
			clamp_temp_variable = { var = max_num_senators min = 0 max = num_senators }
			subtract_from_temp_variable = { max_num_senators = current_support_raw }

			check_variable = { min_num_senators = max_num_senators }
		}
		localization_key = "BRA_rally_effect_dynamic_text_equal"
	}
	text = {
