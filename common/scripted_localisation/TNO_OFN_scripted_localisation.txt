defined_text = {
	name = GetCanadaVote
	text = {
		trigger = {
			CAN = {
				is_subject = no
				has_country_flag = OFN_supports_new_member
			}
		}
		localization_key = OFN_CAN_supports_member
	}
	text = {
		trigger = {
			CAN = {
				is_subject = no
				has_country_flag = OFN_neutral_new_member
			}
		}
		localization_key = OFN_CAN_neutral_member
	}
	text = {
		trigger = {
			CAN = {
				is_subject = no
				has_country_flag = OFN_opposes_new_member
			}
		}
		localization_key = OFN_CAN_opposes_member
	}
	text = { #Falllback case
		trigger = {
			always = yes
		}
		localization_key = OFN_NoText
	}
}
