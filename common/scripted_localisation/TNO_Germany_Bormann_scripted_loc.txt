BOR_GetSIGType = {
	name = BOR_GetSIGType

	text = {
		trigger = { check_variable = { BOR_SIGType^Kartenhaus_SIGGridBox_i = 0 } }
		localization_key = BOR_SIGType_Dismantled
	}

	text = {
		trigger = { check_variable = { BOR_SIGType^Kartenhaus_SIGGridBox_i = 1 } }
		localization_key = BOR_SIGType_Immutable
	}

	text = {
		trigger = {
			check_variable = { BOR_SIGType^Kartenhaus_SIGGridBox_i > 1 }
			check_variable = { BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i }
			check_variable = { BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_Mil^Kartenhaus_SIGGridBox_i }
		}
		localization_key = BOR_SIGType_Allied
	}

	text = {
		trigger = {
			check_variable = { BOR_SIGType^Kartenhaus_SIGGridBox_i > 1 }
			OR = {
				#If Reformists are Supportive
				AND = {
					check_variable = { BOR_ReformistSupport > 49 }
					check_variable = { BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i }

				#If Militarists are Supportive

BOR_GetSIGLoyalty_Top_Faction = {
	name = BOR_GetSIGLoyalty_Top_Faction
	text = {
		trigger = { # Reformist loyalty is greater than Conservative loyalty and Militarist loyalty
			check_variable = { BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i }
			check_variable = { BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_Mil^Kartenhaus_SIGGridBox_i }
		}
		localization_key = BOR_SIGLoyalty_Faction_Ref
	}

	text = {
		trigger = {
			OR = {
				AND = { # Conservative loyalty is greater than Reformist loyalty and Militarist loyalty
					check_variable = { BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i }
					check_variable = { BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_Mil^Kartenhaus_SIGGridBox_i }
				}
				# Special cases where multiple factions have the same loyalty value
				# Preference for the top slot is given to the Conservatives
				AND = { # Conservative loyalty is equal to Reformist loyalty and greater than Militarist loyalty
					check_variable = { BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i = BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i }
					check_variable = { BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_Mil^Kartenhaus_SIGGridBox_i }
				}
				AND = { # Conservative loyalty is equal to Militarist loyalty and greater than Reformist loyalty
					check_variable = { BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i = BOR_SIGLoyalty_Mil^Kartenhaus_SIGGridBox_i }
					check_variable = { BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i }
				}
			}
		}
		localization_key = BOR_SIGLoyalty_Faction_State

		trigger = { # Militarist loyalty is greater than Reformist loyalty and Conservative loyalty

		}
		localization_key = BOR_SIGLoyalty_Faction_Mil
	}
}

BOR_GetSIGLoyalty_Mid_Faction = {
	name = BOR_GetSIGLoyalty_Mid_Faction

	#Let's assume
	#a = Ref Loyalty
	#b = State Loyalty
	#c = Mil Loyalty
	text = {
		trigger = {
			#If true; then we know that Ref Loyalty is the Middle Value
			OR = {
				#b<a & a<c => b<a<c
				AND = {
					check_variable = { BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i < BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i }	#b < a
					check_variable = { BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i < BOR_SIGLoyalty_Mil^Kartenhaus_SIGGridBox_i }		#a < c
				}
				#a<b & c<a => c<a<b
				AND = {
					check_variable = { BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i < BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i }	#a < b
					check_variable = { BOR_SIGLoyalty_Mil^Kartenhaus_SIGGridBox_i < BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i }		#c < a
				}
				#a=b & a<c => a=b<c (preference given to the Conservatives)
				AND = {
					check_variable = { BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i = BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i }	#a = b
					check_variable = { BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i < BOR_SIGLoyalty_Mil^Kartenhaus_SIGGridBox_i }		#a < c
				}
				#a=b & a>c => c<a=b (preference given to the Conservatives)
				AND = {
					check_variable = { BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i = BOR_SIGLoyalty_State^Kartenhaus_SIGGridBox_i }	#a = b
					check_variable = { BOR_SIGLoyalty_Ref^Kartenhaus_SIGGridBox_i > BOR_SIGLoyalty_Mil^Kartenhaus_SIGGridBox_i }		#a > c
				}
				#a=c & a<b => c=a<b (preference given to the Reformists)
