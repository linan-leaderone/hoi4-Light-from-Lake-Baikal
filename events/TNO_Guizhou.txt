country_event = {#Guizhou gets eaten by Yunnan
	id = guz.100
	title = guz.100.t
	desc = guz.100.desc

	picture = GFX_report_event_CHI_nanjing_summit_diktat

	immediate = {
		log = "[GetDateText]: [Root.GetName]: event guz.100"
		clr_global_flag = GUZ_annexation
	}

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = guz.100.a
		if = {
			limit = {
				is_ai = yes
			}
			CHI = { end_puppet = GUZ }
			YUN = {
				add_to_variable = { YUN.chi_influence = GUZ.chi_influence }
				divide_variable = { YUN.chi_influence = 2 }
				round_variable = YUN.chi_influence
				annex_country = {
					target = GUZ
					transfer_troops = yes
				}
				603 = {
					add_core_of = YUN
					remove_core_of = GUZ
				}
				set_cosmetic_tag = YUN_southwest_united
			}
		}
		else = {
			CHI = { end_puppet = GUZ }
			YUN = {
				add_to_variable = { YUN.chi_influence = GUZ.chi_influence }
				divide_variable = { YUN.chi_influence = 2 }
				round_variable = YUN.chi_influence
				annex_country = {
					target = GUZ
					transfer_troops = yes
				}
				603 = {
					add_core_of = YUN
					remove_core_of = GUZ
				}
				set_cosmetic_tag = YUN_southwest_united
				change_tag_from = ROOT
			}
		}
	}
}

country_event = {#Guizhou diplomatically joins Yunnan
	id = guz.102
	title = guz.102.t
	desc = guz.102.desc

	picture = GFX_report_event_CHI_nanjing_summit_diktat

	immediate = {
		log = "[GetDateText]: [Root.GetName]: event guz.102"
		clr_global_flag = GUZ_annexation
	}

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = guz.102.a
		YUN = {
			if = {
				limit = {
					check_variable = { YUN_centralisation_counter < 2 }
				}
				set_country_flag = YUN_decentralized_unification
			}
			else = {
				set_country_flag = YUN_centralized_unification
			}
		}
		if = {
			limit = {
				is_ai = yes
			}
			CHI = { end_puppet = GUZ }
			YUN = {
				add_to_variable = { YUN.chi_influence = GUZ.chi_influence }
				divide_variable = { YUN.chi_influence = 2 }
				round_variable = YUN.chi_influence
				annex_country = {
					target = GUZ
					transfer_troops = yes
				}
				add_state_core = 603
				set_cosmetic_tag = YUN_southwest_united
			}
		}
		else = {
			CHI = { end_puppet = GUZ }
			YUN = {
				add_to_variable = { YUN.chi_influence = GUZ.chi_influence }
				divide_variable = { YUN.chi_influence = 2 }
				round_variable = YUN.chi_influence
				annex_country = {
					target = GUZ
					transfer_troops = yes
				}
				add_state_core = 603
				set_cosmetic_tag = YUN_southwest_united

				change_tag_from = ROOT
			}
		}
	}
}
