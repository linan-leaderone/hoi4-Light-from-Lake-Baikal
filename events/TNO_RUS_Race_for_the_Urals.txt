country_event = { #The Race For The Urals - West Russia
	id = RUS_race_for_the_urals.1
	immediate = {
		hidden_effect = {
			log = "[GetDateText]: [Root.GetName]: event RUS_race_for_the_urals.1"
			custom_effect_tooltip = BEGIN_SOUTHERN_URALS_CRISIS
			set_country_flag = BEGUN_SOUTHERN_URALS_CRISIS
			RUS_urals_influence_initialise = yes
			if = {
				limit = {
					NOT = {
						WSB = { has_country_flag = BEGUN_SOUTHERN_URALS_CRISIS }
					}
				}
				WSB = {
					if = {
						limit = {
							NOT = {
								original_tag = OMS
								has_global_flag = OMS_BatovInsurrectionSuccess
							}
						}
						country_event = { id = RUS_race_for_the_urals.2 }
					}
				}
			}
		}
	}
	trigger = {
		NOT = {
			has_country_flag = BEGUN_SOUTHERN_URALS_CRISIS
		}
	}
	title = RUS_race_for_the_urals.1.t
	desc = RUS_race_for_the_urals.1.d
	picture = GFX_report_event_generic_sign_treaty2

	is_triggered_only = yes

	fire_only_once = yes

	option = {
		name = RUS_race_for_the_urals.1.a
		add_political_power = 25
	}
}

country_event = { #Orenburg - we've been integrated!
	id = RUS_race_for_the_urals.16
	immediate = { log = "[GetDateText]: [Root.GetName]: event RUS_race_for_the_urals.16" }
	title = RUS_race_for_the_urals.16.t
	desc = RUS_race_for_the_urals.16.d
	picture = GFX_report_event_generic_sign_treaty2

	is_triggered_only = yes

	option = { #Accept it
		name = RUS_race_for_the_urals.16.a
		ai_chance = {
			factor = 150
		}
		FROM = {
			country_event = {
				id = RUS_race_for_the_urals.37
				days = 2
			}
		}
	}
	option = { #Resist!
		name = RUS_race_for_the_urals.16.b
		ai_chance = {
			factor = 10
			modifier = {
				factor = 25
				OR = {
					FROM = {
						has_government = fascism
					}
					FROM = {
						has_government = national_socialism
					}
					FROM = {
						has_government = ultranationalism
					}
				}
			}
		}
		set_autonomy = {
			target = ORE
			autonomy_state = autonomy_free
		}
		if = {
			limit = {
				has_global_flag = WSR_INTEGRATING_ORE
			}
			WSB = {
				country_event = {
					id = RUS_race_for_the_urals.19
					days = 2
				}
			}
			WSR = {
				country_event = {
					id = RUS_race_for_the_urals.59
					days = 1
				}
				end_puppet = ORE
			}
			set_variable = { WSR.RUS_urals_relations_level^0 = 1 }
		}
		if = {
			limit = {
				has_global_flag = WSB_INTEGRATING_ORE
			}
			WSR = {
				country_event = {
					id = RUS_race_for_the_urals.19
					days = 1
				}
			}
			WSB = {
				country_event = {
					id = RUS_race_for_the_urals.59
					days = 1
				}
				end_puppet = ORE
			}
			set_variable = { WSB.RUS_urals_relations_level^0 = 1 }
		}
		if = {
			limit = { has_war_with = FIN }
			white_peace = FIN
		}
	}
}
