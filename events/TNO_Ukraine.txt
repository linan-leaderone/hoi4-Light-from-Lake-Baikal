		}
		else = {
			effect_tooltip = {
				UKR_bb_consumption_complete_consumption_cycle_effects = yes
			}
		}

		clr_country_flag = BB_DEMAND_GRAIN_OK
		clr_country_flag = BB_LOCAL_GRAIN_GREAT

		custom_effect_tooltip = UKR.consumption_cycle.a.exploitation_deficit

			array = UKR_bb_map_array
			var:v = {
				multiply_variable = { grain_output = 0.01 }

			}
		}

		UKR_bb_consumption_cycle_dynmod_add = yes
	}

# the nazis don't give up just because you did a sneaky delall
# note: this should NEVER happen in actual gameplay. this should only happen if the player as Ukraine beats Germany or someone thinks they're real clever and uses delall GER to force Ukraine to win
# this event just serves as a way to handle the nuke counter with minimal performance impact while tracking when the nukes are meant to fly
country_event = {
	id = UKR_nuke.0
	immediate = { log = "[GetDateText]: [Root.GetName]: event UKR_nuke.0" }
	title = UKR_nuke.0.t
	desc = UKR_nuke.0.desc
	picture = GFX_report_event_generic_night_sky
	is_triggered_only = yes
	hidden = yes

	option = {
		name = UKR_nuke.0.a
		if = {
			limit = {
				OR = {
					GER = {
						surrender_progress > 0.05
					}
					AND = {
						is_ai = no
						check_variable = { UKR_nuke_counter = 20 }
					}
					AND = {
						is_ai = yes
						check_variable = { UKR_nuke_counter = 45 }
					}
				}
			}
			country_event = {
				id = UKR_nuke.1
				days = 3
			}
			clear_variable = UKR_nuke_counter
		}

		else = {
			add_to_variable = { UKR_nuke_counter = 1 }
			country_event = {
				id = UKR_nuke.0
				days = 7
			}
		}
	}
}
