country_event = { #Tiso dies during the GCW, Å tefan takes over
	id = SLO_skeleton.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event SLO_skeleton.1" }
	title = SLO_skeleton.1.t
	desc = SLO_skeleton.1.desc
	picture = GFX_report_event_SLO_tiso

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = SLO_skeleton.1.a
		retire_character = SLO_Jozef_Tiso
		SLO_Stefan_Tiso = {
			promote_character = {
				ideology = national_socialism_clerical_fascism_subtype
			}
		}
		remove_ideas = SLO_Stefan_Tiso_hog
		add_ideas = SLO_Pavol_Carnogursky_hog
		add_ideas = SLO_tiso_dead
		set_country_flag = SLO_josef_tiso_dead
		hidden_effect = { country_event = { id = SLO_skeleton.2 days = 90 } }
	}
}
