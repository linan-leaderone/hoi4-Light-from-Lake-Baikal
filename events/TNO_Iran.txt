country_event = { #The Coalition Collapses
	id = PER.3
	immediate = { log = "[GetDateText]: [Root.GetName]: event PER.3" }
	title = PER.3.t
	desc = PER.3.d

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = PER.3.a
		remove_faction_status = yes
		every_country = {
			limit = { is_in_faction_with = IRI }
			set_major = no
			remove_faction_status = yes
		}
		IRI = {
			set_major = no
			dismantle_faction = yes
		}
		if = {
			limit = {
				country_exists = IRA
			}
			IRA = {
				declare_war_on = {
					target = IRI
					type = civil_war
				}
				declare_war_on = {
					target = IRL
					type = civil_war
				}
				declare_war_on = {
					target = IRC
					type = civil_war
				}
			}
		}
		if = {
			limit = {
				country_exists = IRI
			}
			IRI = {
				declare_war_on = {
					target = IRL
					type = civil_war
				}
				declare_war_on = {
					target = IRC
					type = civil_war
				}
			}
		}
		if = {
			limit = {
				country_exists = IRL
			}
			IRL = {
				declare_war_on = {
					target = IRC
					type = civil_war
				}
			}
		}
		set_global_flag = TNO_IRA_collapsed_coalition
		every_country = {
			limit = { is_ai = no }
			news_event = {
				id = WORLD.652
				days = 1
			}
		}
		if = {
			limit = {
				country_exists = PER
			}
			PER = {
				every_enemy_country = {
					limit = {
						has_capitulated = yes
					}
					if = {
						limit = { tag = IRL }
						USA = { country_event = USA_IR.6 }
					}
					else_if = {
						limit = { tag = IRI }
						JAP = { country_event = jap_fopo.2 }
					}

					remove_from_array = { global.PER_contenders = THIS }
					PER = {
						white_peace = PREV
						annex_country = {
							target = PREV
							transfer_troops = no
						}
					}
				}
			}
		}
						
	}
